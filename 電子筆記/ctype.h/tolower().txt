#include <stdio.h>
#include <ctype.h>
#include <string.h>
int main ()
{
    char str[]="ABCDEFG";
    int x=strlen(str);
    for(int i=0;i<x;i++){
        str[i]=tolower(str[i]);
        printf("%c",str[i]);
    }
}
tolower(x)此函數會將x字母轉換成小寫字母(如果x為大寫字母的話)
上面的程式是把str字串裡的所有大寫字母轉換成小寫字母，並印出。
